§ 01

클로져란 무엇인가 — 기본 개념

클로져(Closure)는 주변 환경(스코프)의 변수를 캡처(capture)할 수 있는 익명 함수입니다. The Rust Book은 클로져를 "anonymous functions you can save in a variable or pass as arguments to other functions"로 정의합니다. 일반 fn과 달리 자신이 정의된 스코프의 변수를 기억합니다.

문법
|params| { body }
매개변수·반환 타입 모두 생략 가능 (컴파일러 추론)
vs 일반 함수
환경 변수 캡처 가능
fn은 외부 변수를 직접 캡처할 수 없음
캡처 방식
컴파일러가 자동 추론
Copy / &T / &mut T / 소유권 이전 순
move 키워드
소유권 강제 이전
스레드 전달 시 필수 (Send + 'static)
closure_basics.rs 기본 
// 클로져의 가장 기본적인 형태와 일반 함수 비교

// ① 일반 함수 — 외부 변수를 직접 참조할 수 없음
fn add_fn(x: i32, y: i32) -> i32 { x + y }

fn main() {
    // ② 클로져 기본 — 타입 완전 생략 (컴파일러가 사용처에서 추론)
    let add = |x, y| x + y;
    println!("add(3,4) = {}", add(3, 4)); // 7

    // ③ 클로져 — 타입 명시 (선택사항)
    let add_typed = |x: i32, y: i32| -> i32 { x + y };

    // ④ 환경 캡처 — 클로져만의 특권
    let offset = 100; // 외부 변수
    let add_offset = |x| x + offset; // offset을 캡처!
    println!("add_offset(5) = {}", add_offset(5)); // 105
    println!("offset 여전히 유효: {}", offset);         // 100 (불변 참조 캡처)

    // ⑤ 클로져를 변수에 저장하고 나중에 호출
    let expensive = |n: u64| -> u64 {
        // 실제 코드에서는 오래 걸리는 연산이 들어갈 수 있음
        n * n
    };
    let result = expensive(10);
    println!("expensive(10) = {}", result); // 100

    // ⑥ 클로져를 다른 함수의 인자로 전달
    let nums = vec![1, 2, 3, 4, 5];
    let evens: Vec<&i32> = nums.iter().filter(|&&x| x % 2 == 0).collect();
    println!("짝수: {:?}", evens); // [2, 4]
}
📌 The Rust Book (ch.13) 정의
클로져는 변수에 저장하거나 다른 함수의 인자로 전달할 수 있는 익명 함수입니다. 함수와 달리 클로져는 자신이 정의된 스코프의 값들을 캡처할 수 있습니다.
§ 02

캡처 방식 — 컴파일러 추론 규칙

컴파일러는 클로져 본문에서 변수가 어떻게 사용되는지 분석하여 가장 제한적인(least restrictive) 캡처 방식을 자동으로 선택합니다. 읽기만 하면 불변 참조, 수정하면 가변 참조, 소유권을 이동시키면 소유권 캡처 순서로 결정됩니다. move 키워드를 사용하면 이 추론을 무시하고 모든 변수를 소유권 이전으로 강제합니다.

⚠️ 중요: 캡처 방식 결정 우선순위
컴파일러는 ① 불변 참조(&T) → ② 가변 참조(&mut T) → ③ 소유권 이전(move) 순서로 가능한 가장 약한 방식을 선택합니다. move를 명시하면 Copy 타입은 복사, 나머지는 소유권 이전이 됩니다.
capture_modes.rs 캡처 방식 
// 세 가지 캡처 방식을 모두 보여주는 예제

fn main() {
    // ━━━━ ① 불변 참조(&T) 캡처 ━━━━
    // 클로져 내부에서 변수를 읽기만 할 때
    let text = String::from("Hello");
    let print_text = || println!("{}", text); // &text 로 캡처 → Fn

    print_text(); // Hello
    print_text(); // Hello (Fn이므로 여러 번 호출 가능)
    println!("text 여전히 유효: {}", text); // ✅ 불변 참조이므로 원본 접근 OK

    // ━━━━ ② 가변 참조(&mut T) 캡처 ━━━━
    // 클로져 내부에서 변수를 수정할 때
    let mut count = 0;
    let mut inc = || {
        count += 1; // count를 수정 → &mut count 로 캡처 → FnMut
        count
    };

    println!("inc() = {}", inc()); // 1
    println!("inc() = {}", inc()); // 2
    // println!("{}", count); // ❌ 클로져가 살아있는 동안 가변 빌림 중
    drop(inc); // 클로져 드롭 → 빌림 해제
    println!("count = {}", count); // ✅ 2

    // ━━━━ ③ 소유권 이전(move) 캡처 ━━━━
    // 클로져 내부에서 소유권을 소비하는 연산을 할 때,
    // 또는 move 키워드를 명시할 때
    let name = String::from("Rust");
    let consume = move || {
        // move로 name의 소유권이 클로져 안으로 이동
        let upper = name.to_uppercase(); // name 소유
        println!("upper: {}", upper);
    };
    consume();
    // println!("{}", name); // ❌ 소유권이 클로져로 이동됨

    // ━━━━ ④ Copy 타입은 move해도 원본 유효 ━━━━
    let x: i32 = 42; // i32는 Copy 트레이트 구현
    let cl = move || println!("x = {}", x); // x의 복사본을 캡처
    cl();
    println!("x 여전히 유효: {}", x); // ✅ Copy이므로 원본 유효
}
캡처 방식클로져 내부 동작자동 구현 트레이트원본 접근
&T (불변 참조)읽기만 함Fn FnMut FnOnce✅ 가능
&mut T (가변 참조)수정함FnMut FnOnce❌ 빌림 중
소유권 이전drop/into_* 등 소비FnOnce❌ 이동됨
move + Copy 타입값 복사 후 사용Fn (상황에 따라)✅ 복사본 남음
§ 03

클로져 내부 구조 — 컴파일러가 생성하는 익명 구조체

Rust의 클로져는 컴파일 시점에 캡처한 변수들을 필드로 갖는 익명 구조체(anonymous struct)로 변환됩니다. 그리고 그 구조체에 캡처 방식에 따라 Fn, FnMut, FnOnce 중 적절한 트레이트가 자동 구현됩니다. 이 개념을 이해하면 왜 특정 규칙이 존재하는지 명확히 알 수 있습니다.

closure_desugared.rs 내부 구조 
// 클로져가 내부적으로 어떻게 변환되는지 보여주는 예시
// (실제 Rust 코드는 아니지만, 컴파일러가 하는 일을 설명)

// ━━ 우리가 작성하는 코드 ━━
let x = 5;
let y = 10;
let add_xy = |n| n + x + y; // x, y를 불변 참조로 캡처

// ━━ 컴파일러가 내부적으로 생성하는 구조체 (개념 설명용) ━━
// struct ClosureAddXY<'a> {
//     x: &'a i32,   // 불변 참조로 캡처
//     y: &'a i32,
// }
// impl<'a> Fn(i32) -> i32 for ClosureAddXY<'a> {
//     fn call(&self, n: i32) -> i32 {
//         n + *self.x + *self.y
//     }
// }

// ━━ 가변 캡처 예시 ━━
let mut total = 0;
let mut accumulate = |n| { total += n; }; // total을 &mut 로 캡처
// 컴파일러가 생성하는 구조체:
// struct ClosureAccumulate<'a> {
//     total: &'a mut i32,  // 가변 참조로 캡처
// }
// impl<'a> FnMut(i32) for ClosureAccumulate<'a> {
//     fn call_mut(&mut self, n: i32) {
//         *self.total += n;
//     }
// }

// ━━ 실제로 활용되는 패턴: struct에 클로져 저장 ━━
struct Cacher<T>
where
    T: Fn(u32) -> u32, // T는 Fn 트레이트를 구현해야 함
{
    calculation: T,                  // 클로져를 제네릭 필드로 저장
    value: Option<u32>,            // 계산 결과를 캐시
}

impl<T> Cacher<T>
where
    T: Fn(u32) -> u32,
{
    fn new(calculation: T) -> Cacher<T> {
        Cacher { calculation, value: None }
    }

    // 처음 호출 시 계산, 이후 캐시된 값 반환 (Lazy evaluation)
    fn value(&mut self, arg: u32) -> u32 {
        match self.value {
            Some(v) => v,
            None => {
                let v = (self.calculation)(arg); // 클로져 호출
                self.value = Some(v);
                v
            }
        }
    }
}

fn main() {
    let mut expensive_result = Cacher::new(|num| {
        println!("계산 중... (오래 걸리는 작업)");
        num * num + num
    });

    println!("결과1: {}", expensive_result.value(5)); // 계산 실행됨 → 30
    println!("결과2: {}", expensive_result.value(5)); // 캐시 반환 → 30 (재계산 없음)
}
✅ 핵심 포인트
클로져는 실제로는 캡처한 변수들을 필드로 가진 익명 struct입니다. Fn/FnMut/FnOnce 트레이트는 이 struct에 구현됩니다. 따라서 클로져를 struct 필드, Vec, HashMap에 저장하거나 함수의 반환값으로 사용할 수 있습니다.
§ 04

트레이트 시그니처 — std::ops 공식 정의

세 트레이트의 핵심 차이는 self 매개변수의 형태에 있습니다. call은 &self, call_mut은 &mut self, call_once는 self를 받습니다. 이 차이가 반복 호출 가능 여부와 원본 변수 수정 가능 여부를 결정합니다.

trait_signatures.rs (출처: std::ops, Rust 1.x) 공식 정의 
// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// 출처: https://doc.rust-lang.org/std/ops/trait.FnOnce.html
//       https://doc.rust-lang.org/std/ops/trait.FnMut.html
//       https://doc.rust-lang.org/std/ops/trait.Fn.html
// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// ① FnOnce — 최상위 트레이트 (모든 클로져가 구현)
pub trait FnOnce<Args> {
    type Output;
    // self를 소비 → 딱 한 번만 호출 가능
    fn call_once(self, args: Args) -> Self::Output;
}

// ② FnMut — FnOnce의 서브트레이트
pub trait FnMut<Args>: FnOnce<Args> {
    // &mut self → 상태 변경 가능, 반복 호출 가능
    fn call_mut(&mut self, args: Args) -> Self::Output;
}

// ③ Fn — FnMut의 서브트레이트 (가장 엄격한 요구사항)
pub trait Fn<Args>: FnMut<Args> {
    // &self → 상태 변경 불가, 반복 호출 가능, 공유 참조로 호출
    fn call(&self, args: Args) -> Self::Output;
}

// ━━ 자동 구현 규칙 (표준 라이브러리 보장) ━━
// - Fn 구현 → FnMut, FnOnce 자동 구현
// - FnMut 구현 → FnOnce 자동 구현
// - 역방향은 성립하지 않음
//
// 추가 보장 (std::ops 문서):
// - &F가 Fn 구현 → &F도 FnMut 구현 (Fn이 FnMut의 subtrait이므로)
// - &mut F가 FnMut 구현 → &mut F로도 FnMut 호출 가능
// - Box<F>가 FnMut 구현 → Box<dyn FnMut> 가능

fn main() {
    // self 수신 방식을 직접 확인하는 예시
    let x = 10;

    // Fn → &self 로 호출 → 공유 참조(&F)로도 호출 가능
    let fn_cl = || println!("x = {}", x);
    let r = &fn_cl; // &F
    r();            // &F도 Fn 구현 → 호출 가능
    fn_cl();        // 원본도 그대로

    // FnMut → &mut F 로도 호출 가능
    let mut n = 0;
    let mut fnmut_cl = || { n += 1; println!("n = {}", n); };
    {
        let rm = &mut fnmut_cl; // &mut F
        rm();                    // &mut F도 FnMut 구현 → 호출 가능
    }
    fnmut_cl(); // n = 2
}
📌 서브트레이트 관계의 의미
Fn: FnMut는 "Fn은 FnMut의 서브트레이트"를 의미합니다. 즉, Fn을 구현하려면 반드시 FnMut도 구현해야 합니다. 따라서 FnMut를 요구하는 곳에 Fn을 전달할 수 있습니다 (리스코프 치환 원칙과 유사). 하지만 반대로 Fn을 요구하는 곳에 FnMut만 구현한 클로져는 전달할 수 없습니다.

§ 06

Fn 트레이트 — 예제 5개

Fn은 캡처한 변수를 불변 참조(&T)로만 사용합니다. &self로 호출되므로 몇 번이든 반복 호출 가능하며, 동시에 여러 곳에서 사용할 수도 있습니다. 표준 라이브러리의 map, filter 등 대부분의 Iterator 어댑터가 Fn을 요구합니다.

📌 std::ops::Fn 공식 정의
"An instance of Fn can be called repeatedly without mutating state." — Fn을 구현하는 타입에는 모든 safe 함수 포인터도 포함됩니다. &F가 Fn을 구현하면 &F도 자동으로 FnMut를 구현합니다 (std::ops 문서 명시).
Fn예제 1 — 기본: 불변 참조 캡처, 반복 호출, 원본 변수 유지&self · 기본
fn_01_basic.rsFn
// ─────────────────────────────────────────────────────────
// [Fn 예제 1] 기본: 불변 참조(&T) 캡처
// 규칙: 클로져 내부에서 변수를 읽기만 하면 자동으로 Fn
// &self 호출 → 몇 번이든 반복 가능, 원본 변수 유지
// ─────────────────────────────────────────────────────────

// Fn 바운드: f를 몇 번이든 반복 호출 가능
fn call_twice<F: Fn(i32) -> i32>(f: F, val: i32) -> (i32, i32) {
    (f(val), f(val)) // &self 호출이므로 두 번 다 OK
}

fn main() {
    // ① String을 불변 참조로 캡처
    let greeting = String::from("안녕하세요");
    let greet = |name: &str| format!("{}, {}!", greeting, name);

    println!("{}", greet("Alice")); // 안녕하세요, Alice!
    println!("{}", greet("Bob"));   // 안녕하세요, Bob! (Fn → 재호출 OK)
    println!("greeting 유효: {}", greeting); // ✅ 불변 참조이므로 원본 유지

    // ② i32(Copy 타입)는 값이 복사되어 캡처
    let base = 100;
    let add_base = |x| x + base; // base의 복사본 캡처 → Fn
    let (r1, r2) = call_twice(add_base, 5);
    println!("r1={}, r2={}", r1, r2); // 105, 105
    println!("base 유효: {}", base);      // ✅ 여전히 100

    // ③ 여러 변수를 동시에 캡처 (모두 불변 참조)
    let prefix = String::from("LOG");
    let level = 3;
    let log = |msg: &str| format!("[{}][Lv{}] {}", prefix, level, msg);
    println!("{}", log("서버 시작"));
    println!("prefix 유효: {}", prefix); // ✅

    // ④ Fn은 공유 참조(&F)로도 호출 가능 (std::ops 문서 명시)
    let double = |x: i32| x * 2;
    let r = &double;
    println!("r(7) = {}", r(7)); // 14 — &F도 Fn 구현

Fn예제 3 — fn 포인터 호환 + 함수 합성 compose제네릭 + 합성

// ─────────────────────────────────────────────────────────
// [Fn 예제 3] fn 포인터 호환 + compose
// std::ops 문서: "safe function pointers implement Fn"
// fn 포인터는 Fn, FnMut, FnOnce 모두 자동 구현
// ─────────────────────────────────────────────────────────

fn apply_n_times<T: Clone, F: Fn(T) -> T>(f: F, mut val: T, n: usize) -> T {
    for _ in 0..n { val = f(val.clone()); }
    val
}

fn compose<T, F1: Fn(T)->T, F2: Fn(T)->T>(f1: F1, f2: F2) -> impl Fn(T) -> T {
    move |x| f2(f1(x))
}

fn double(x: i32) -> i32 { x * 2 }
fn square(x: i32) -> i32 { x * x }

fn main() {
    // fn 포인터 전달 (Fn 자동 구현)
    println!("1을 4번 double: {}", apply_n_times(double, 1, 4)); // 16
    println!("2를 3번 square: {}", apply_n_times(square, 2, 3)); // 256

    // 클로져 전달
    println!("0에 3을 5번: {}", apply_n_times(|x| x + 3, 0, 5)); // 15

    // 함수 합성
    let double_then_square = compose(double, square);
    println!("double_then_square(3) = {}", double_then_square(3)); // (3*2)^2=36
    println!("double_then_square(5) = {}", double_then_square(5)); // (5*2)^2=100

    // fn 포인터는 같은 타입 → Vec에 직접 저장 가능
    let ops: Vec<fn(i32) -> i32> = vec![double, square];
    let mut val = 3;
    for op in &ops { val = op(val); } // 3 → 6 → 36
    println!("파이프라인: {}", val);
}

Fn예제 5 — struct에 Fn 저장: EventBus + 변환 PipelineBox<dyn Fn>
fn_05_struct_store.rsFn
// ─────────────────────────────────────────────────────────
// [Fn 예제 5] struct에 Fn 저장
// Box<dyn Fn> → 서로 다른 타입의 클로져를 하나의 Vec에 저장
// 'static: 클로져가 스택 프레임 수명을 초과해도 유효하도록
// ─────────────────────────────────────────────────────────

struct EventBus {
    handlers: Vec<Box<dyn Fn(&str)>>,
}
impl EventBus {
    fn new() -> Self { EventBus { handlers: Vec::new() } }
    fn on<F: Fn(&str) + 'static>(&mut self, h: F) { self.handlers.push(Box::new(h)); }
    fn emit(&self, ev: &str) { for h in &self.handlers { h(ev); } }
}

struct Pipeline {
    steps: Vec<Box<dyn Fn(String) -> String>>,
}
impl Pipeline {
    fn new() -> Self { Pipeline { steps: Vec::new() } }
    fn step<F: Fn(String)->String+'static>(&mut self, f: F) -> &mut Self {
        self.steps.push(Box::new(f)); self
    }
    fn run(&self, input: String) -> String {
        self.steps.iter().fold(input, |acc, s| s(acc))
    }
}

fn main() {
    let mut bus = EventBus::new();
    bus.on(|e| println!("[Logger]  {}", e));
    bus.on(|e| println!("[Monitor] {}", e));
    bus.emit("사용자 로그인");
    bus.emit("데이터 저장");

    let mut p = Pipeline::new();
    p.step(|s| s.trim().to_string())
     .step(|s| s.to_lowercase())
     .step(|s| s.replace(' ', "_"));
    println!("결과: '{}'", p.run("  Hello World  ".to_string())); // 'hello_world'
}

FnMut 트레이트 — 예제 5개

FnMut는 캡처한 변수를 가변 참조(&mut T)로 빌립니다. &mut self로 호출되므로 반복 호출은 가능하지만, 클로져가 살아있는 동안 캡처한 변수를 직접 사용할 수 없습니다. 클로져 변수를 mut로 선언해야 하며, 함수 인자로 받을 때도 mut f: F로 받아야 합니다.

📌 std::ops::FnMut 공식 정의
"FnMut is implemented automatically by closures which take mutable references to captured variables." 또한 &mut F가 FnMut를 구현하면 &mut F도 FnMut를 자동 구현합니다. sort_by_key처럼 한 항목당 여러 번 클로져를 호출해야 하는 경우에 사용합니다.

// ─────────────────────────────────────────────────────────
// [FnMut 예제 1] 카운터 — 가변 참조(&mut T) 캡처
// 핵심: 클로져 내부에서 변수를 '수정'하면 자동 FnMut
// 클로져 변수는 반드시 mut, 인자로 받을 때도 mut f 필수
// ─────────────────────────────────────────────────────────

// FnMut 바운드: mut f 로 받아야 &mut self 호출 가능
fn call_three_times<F: FnMut()>(mut f: F) {
    f(); f(); f();
}

fn main() {
    let mut count = 0;
    let mut counter = || {
        count += 1; // count를 수정 → &mut count 캡처 → FnMut
        println!("카운트: {}", count);
    };
    counter(); counter(); counter();
    // println!("{}", count); // ❌ 가변 빌림 중
    drop(counter);
    println!("최종: {}", count); // ✅ 3

    // call_three_times에 FnMut 전달
    let mut log = Vec::new();
    call_three_times(|| log.push("이벤트")); // log 수정 → FnMut
    println!("log: {:?}", log); // ["이벤트", "이벤트", "이벤트"]

    // &mut F도 FnMut 구현 (std::ops 문서)
    let mut n = 0;
    let mut inc = || { n += 1; };
    {
        let rm = &mut inc;
        rm(); rm(); // &mut F로도 호출 가능
    }
    drop(inc);
    println!("n = {}", n); // 2
}

// ─────────────────────────────────────────────────────────
// [FnMut 예제 2] 상태를 가진 클로져 반환 — 누산기 / 이동 평균
// impl FnMut 반환 → 각 인스턴스가 독립적인 내부 상태 보유

fn make_accumulator(start: i32) -> impl FnMut(i32) -> i32 {
    let mut total = start;
    move |x| { total += x; total }
}

fn make_moving_avg(window: usize) -> impl FnMut(f64) -> f64 {
    let mut buf: Vec<f64> = Vec::new();
    move |v| {
        buf.push(v);
        let slice = if buf.len() > window { &buf[buf.len()-window..] } else { &buf };
        slice.iter().sum::<f64>() / slice.len() as f64
    }
}

fn make_rate_limiter(max_per_sec: u32) -> impl FnMut() -> bool {
    let mut calls = 0;
    move || { calls += 1; calls <= max_per_sec }
}

fn main() {
    let mut a = make_accumulator(0);
    let mut b = make_accumulator(100); // 독립적 상태
    println!("a: {} {} {}", a(5), a(10), a(3)); // 5 15 18
    println!("b: {} {} {}", b(5), b(10), b(3)); // 105 115 118

    let mut avg3 = make_moving_avg(3);
    for &v in &[10.0, 20.0, 30.0, 40.0, 50.0] {
        println!("avg3({}) = {:.1}", v, avg3(v));
    }

    for i in 1..=5 {
        println!("요청 {}: {}", i, if limiter() { "허용" } else { "거절" });
    }
}

FnMut예제 5 — 상태 머신: FnMut로 전환 로직 구현상태 머신

// ─────────────────────────────────────────────────────────
// [FnMut 예제 5] 상태 머신: FnMut로 전환 로직
// 내부 상태(state)를 갖고 호출마다 전환 → FnMut
// ─────────────────────────────────────────────────────────

#[derive(Debug, Clone, PartialEq)]
enum State { Idle, Running, Paused, Stopped }

fn make_state_machine() -> impl FnMut(&str) -> State {
    let mut state = State::Idle;
    move |cmd| {
        state = match (&state, cmd) {
            (State::Idle,    "start")  => State::Running,
            (State::Running, "pause")  => State::Paused,
            (State::Paused,  "resume") => State::Running,
            (State::Running, "stop")   => State::Stopped,
            (State::Paused,  "stop")   => State::Stopped,
            _                             => state.clone(),
        };
        state.clone()
    }
}

fn make_toggle() -> impl FnMut() -> bool {
    let mut on = false;
    move || { on = !on; on }
}

fn main() {
    let mut sm = make_state_machine();
    for cmd in &["start", "pause", "resume", "stop"] {
        println!("cmd='{}' → {:?}", cmd, sm(cmd));
    }

    let mut toggle = make_toggle();
    for _ in 0..4 { print!("{} ", toggle()); }
    println!(); // true false true false
}

FnOnce는 self를 소비하며 호출됩니다. 클로져 내부에서 캡처한 값의 소유권을 이동시키는 연산(drop(), into_bytes(), 값을 다른 변수로 이동 등)을 하면 자동으로 FnOnce만 구현됩니다. 모든 클로져는 최소한 FnOnce를 구현합니다 (The Rust Book ch.13).

📌 The Rust Book ch.13 원문
"FnOnce applies to closures that can be called once. All closures implement at least this trait because all closures can be called. A closure that moves captured values out of its body will only implement FnOnce and none of the other Fn traits because it can only be called once."

// ─────────────────────────────────────────────────────────
// [FnOnce 예제 2] thread::spawn — FnOnce + Send + 'static
// thread::spawn 시그니처:
//   pub fn spawn<F, T>(f: F) → JoinHandle<T>
//   where F: FnOnce() → T + Send + 'static
// 스레드는 클로져를 딱 한 번 실행하고 종료 → FnOnce
// Send: 다른 스레드로 전달 가능 (Rc<T> 등 불가)
// 'static: dangling reference 방지
// ─────────────────────────────────────────────────────────

use std::thread;
use std::sync::mpsc;

fn main() {
    // ① 기본: 소유권을 스레드로 이동
    let data = vec![1, 2, 3, 4, 5];
    let h1 = thread::spawn(move || {
        // data의 소유권이 이 클로져(스레드)로 이동 → FnOnce
        let sum: i32 = data.iter().sum();
        println!("[스레드1] 합계: {}", sum);
        sum
    });
    println!("스레드1 결과: {}", h1.join().unwrap());
    // println!("{:?}", data); // ❌ 소유권 이동됨

    // ② 채널(mpsc): 스레드 간 결과 전달
    let (tx, rx) = mpsc::channel::<String>();
    let config = String::from("host=localhost:5432");
    thread::spawn(move || {
        // config 소유권 이동 → FnOnce
        let parsed = config.split('=').nth(1).unwrap_or("").to_string();
        tx.send(parsed).unwrap();
    });
    println!("DB 주소: {}", rx.recv().unwrap());

    // ③ 여러 스레드 — 각각 독립적인 클로져 (FnOnce 인스턴스)
    let handles: Vec<_> = (0..4).map(|i| {
        thread::spawn(move || { // i(Copy)는 복사본 캡처
            println!("스레드 {}: 작업 완료", i);
            i * i
        })
    }).collect();
    let results: Vec<i32> = handles.into_iter().map(|h| h.join().unwrap()).collect();
    println!("결과: {:?}", results);
}
FnOnce예제 3 — unwrap_or_else / get_or_insert_with: The Book 예제Option 활용
fnonce_03_option.rsFnOnce
// ─────────────────────────────────────────────────────────
// [FnOnce 예제 3] Option 메서드들의 FnOnce 활용
// unwrap_or_else, get_or_insert_with, ok_or_else 등
// None/Err 일 때만 클로져 호출 → 최대 1회 → FnOnce 충분
// The Rust Book: "unwrap_or_else accepts all three kinds of closures
// and is as flexible as it can be"
// ─────────────────────────────────────────────────────────

// Option::unwrap_or_else 시그니처:
// pub fn unwrap_or_else<F: FnOnce() -> T>(self, f: F) -> T

fn find_user(id: u32) -> Option<String> {
    if id == 1 { Some("Alice".to_string()) } else { None }
}

fn main() {
    // ① unwrap_or_else — None일 때만 클로져 실행 (FnOnce)
    let user = find_user(1).unwrap_or_else(|| "Anonymous".to_string());
    println!("user: {}", user); // Alice
    let guest = find_user(99).unwrap_or_else(|| "Guest".to_string());
    println!("guest: {}", guest); // Guest

    // ② 소유권을 가진 기본값 생성 — FnOnce 필수
    let default_name = String::from("기본 사용자");
    let result = find_user(99).unwrap_or_else(move || default_name);
    println!("result: {}", result);

    // ③ get_or_insert_with — None이면 클로져로 초기화
    let mut cache: Option<Vec<i32>> = None;
    let v = cache.get_or_insert_with(|| {
        println!("초기화 실행 (1회만)");
        vec![1, 2, 3]
    });
    v.push(4);
    println!("cache: {:?}", cache); // Some([1,2,3,4])
    cache.get_or_insert_with(|| {
        println!("이 메시지는 출력 안 됨"); // Some이면 호출 안 함
        vec![]
    });

    // ④ ok_or_else — None을 Err로 변환 (FnOnce)
    let val: Option<i32> = None;
    let err_msg = String::from("값이 없습니다");
    let res: Result<i32, String> = val.ok_or_else(move || err_msg);
    println!("결과: {:?}", res); // Err("값이 없습니다")
}
FnOnce예제 4 — 빌더 패턴: 일회성 초기화 콜백빌더 패턴
fnonce_04_builder.rsFnOnce
// ─────────────────────────────────────────────────────────
// [FnOnce 예제 4] 빌더 패턴 — 일회성 초기화 콜백
// build() 메서드에서 FnOnce를 받아 초기화 로직을 실행
// 초기화는 딱 한 번만 하면 되므로 FnOnce 적합
// ─────────────────────────────────────────────────────────

struct Server { host: String, port: u16, tls: bool }

struct ServerBuilder { host: String, port: u16, tls: bool }

impl ServerBuilder {
    fn new() -> Self {
        ServerBuilder { host: String::from("localhost"), port: 8080, tls: false }
    }
    fn with_config<F: FnOnce(&mut Self)>(mut self, f: F) -> Self {
        f(&mut self); // 초기화 클로져 실행 (1회)
        self
    }
    fn build(self) -> Server {
        Server { host: self.host, port: self.port, tls: self.tls }
    }
}

fn with_callback<T, F: FnOnce() -> T>(produce: F) -> T {
    produce() // 딱 한 번 호출 → FnOnce
}

fn main() {
    let srv = ServerBuilder::new()
        .with_config(|b| {
            b.host = String::from("0.0.0.0");
            b.port = 443;
            b.tls = true;
        })
        .build();
    println!("서버: {}:{} TLS={}", srv.host, srv.port, srv.tls);

    // FnOnce 콜백으로 값 생성
    let value = with_callback(|| {
        println!("값 생성 중...");
        vec![1, 2, 3]
    });
    println!("생성된 값: {:?}", value);
}
FnOnce예제 5 — 계층 실증: Fn ⊆ FnMut ⊆ FnOnce 확인계층 실증
fnonce_05_hierarchy.rsFnOnce
// ─────────────────────────────────────────────────────────
// [FnOnce 예제 5] 트레이트 계층 실증 — 자동 구현 관계
// Fn ⊆ FnMut ⊆ FnOnce
// Fn 클로져 → FnMut, FnOnce 모두 충족
// FnMut 클로져 → FnOnce 충족, Fn은 미충족
// FnOnce 클로져 → FnOnce만 충족
// ─────────────────────────────────────────────────────────

fn needs_fn    <F: Fn()    >(f: F)       { f(); f(); println!("Fn: 2회 완료");     }
fn needs_fn_mut<F: FnMut() >(mut f: F) { f(); f(); println!("FnMut: 2회 완료");  }
fn needs_fn_once<F: FnOnce()>(f: F)    { f();     println!("FnOnce: 1회 완료"); }

fn main() {
    println!("━━ Fn 클로져 ━━");
    let x = 42;
    let fn_cl = || println!("  x={}", x); // &x → Fn
    needs_fn(fn_cl);       // ✅ Fn    → Fn
    needs_fn_mut(fn_cl);   // ✅ Fn    → FnMut  (자동 구현)
    needs_fn_once(fn_cl);  // ✅ Fn    → FnOnce (자동 구현)

    println!("━━ FnMut 클로져 ━━");
    let mut n = 0;
    let mut fnmut_cl = || { n += 1; println!("  n={}", n); };
    // needs_fn(fnmut_cl); // ❌ FnMut는 Fn 미구현
    needs_fn_mut(&mut fnmut_cl); // ✅ FnMut → FnMut
    needs_fn_once(fnmut_cl);      // ✅ FnMut → FnOnce (자동 구현)
    drop(n);

    println!("━━ FnOnce 클로져 ━━");
    let owned = String::from("소유값");
    let fnonce_cl = move || {
        drop(owned); // 소비 → FnOnce만
        println!("  소유권 소비");
    };
    // needs_fn(fnonce_cl);     // ❌
    // needs_fn_mut(fnonce_cl); // ❌
    needs_fn_once(fnonce_cl);     // ✅ FnOnce → FnOnce
}

sort_by_key 에러 케이스 — The Rust Book 실제 예제

The Rust Book ch.13의 핵심 예제입니다. sort_by_key는 FnMut를 요구하는데, 클로져 내부에서 값을 이동(move out)시키면 FnOnce가 되어 컴파일 에러가 납니다. 에러 원인과 해결법을 함께 보여줍니다.

sort_by_key_error.rs — 컴파일 에러 예시 (The Book Listing 13-8)컴파일 에러
// ─────────────────────────────────────────────────────────
// 출처: The Rust Book ch.13 Listing 13-8
// sort_by_key는 FnMut 요구 (각 요소마다 여러 번 호출)
// 아래 클로져는 value를 이동(push by value)시키므로
// FnOnce가 되어 에러 발생
// ─────────────────────────────────────────────────────────

#[derive(Debug)]
struct Rectangle { width: u32, height: u32 }

fn main() {
    let mut list = vec![
        Rectangle { width: 10, height: 1  },
        Rectangle { width: 3,  height: 5  },
        Rectangle { width: 7,  height: 12 },
    ];
    let mut sort_operations: Vec<String> = vec![];
    let value = String::from("closure called");

    // ❌ 에러: sort_by_key는 FnMut 요구, 아래는 FnOnce
    // sort_operations.push(value)는 value를 이동시킴
    // → 클로져가 FnOnce만 구현 → FnMut 불충족
    //
    // error[E0507]: cannot move out of `value`, a captured variable
    //   in an `FnMut` closure
    /*
    list.sort_by_key(|r| {
        sort_operations.push(value); // ← value 이동 발생
        r.width
    });
    */
}
❌ 에러 원인
sort_operations.push(value)는 value의 소유권을 sort_operations로 이동시킵니다. 이동은 한 번만 가능하므로 이 클로져는 FnOnce만 구현합니다. 그런데 sort_by_key는 N개 요소를 정렬하기 위해 클로져를 여러 번 호출하므로 FnMut가 필요합니다.
sort_by_key_fixed.rs — 해결: clone() 사용수정됨
// ─────────────────────────────────────────────────────────
// 해결: value를 이동시키지 않고 clone()으로 복사본 사용
// clone()은 value를 소비하지 않으므로 여러 번 호출 가능
// → FnMut 충족
//
// The Book: "consider cloning the value if the performance
// cost is acceptable"
// ─────────────────────────────────────────────────────────

#[derive(Debug)]
struct Rectangle { width: u32, height: u32 }

fn main() {
    let mut list = vec![
        Rectangle { width: 10, height: 1  },
        Rectangle { width: 3,  height: 5  },
        Rectangle { width: 7,  height: 12 },
    ];
    let mut sort_operations: Vec<String> = vec![];
    let value = String::from("closure called");

    // ✅ 수정: value.clone()으로 복사본을 push
    // value 자체는 이동되지 않으므로 FnMut 충족
    list.sort_by_key(|r| {
        sort_operations.push(value.clone()); // clone() → FnMut OK
        r.width
    });

    println!("{:#?}", list);
    println!("sort_operations: {:?}", sort_operations);
    // sort_operations에는 정렬 비교 횟수만큼 "closure called"가 추가됨

    // 대안: usize 카운터 사용 (Copy 타입 → 이동 없음)
    let mut call_count = 0_usize;
    list.sort_by_key(|r| {
        call_count += 1; // Copy 타입 수정 → FnMut OK
        r.width
    });
    println!("key 함수 호출 횟수: {}", call_count);
}

move 키워드와 Copy 타입

move 키워드는 클로져가 환경 변수를 캡처할 때 모두 소유권 이전으로 강제합니다. 단, Copy 트레이트를 구현하는 타입(i32, f64, bool, char 등)은 소유권 이전 대신 값이 복사됩니다. 따라서 move를 사용해도 Copy 타입의 원본은 그대로 사용할 수 있습니다.

move_and_copy.rsmove + Copy
// ─────────────────────────────────────────────────────────
// move 키워드 동작 규칙:
//   - Copy 타입 (i32, f64, bool, char, &T 등):
//       값이 복사됨 → 원본 변수 여전히 유효
//   - non-Copy 타입 (String, Vec, Box 등):
//       소유권이 클로져로 이전 → 원본 사용 불가
// ─────────────────────────────────────────────────────────

use std::thread;

fn main() {
    // ━━ Copy 타입 ━━
    let x: i32  = 42;
    let flag: bool = true;

    let cl = move || println!("x={}, flag={}", x, flag);
    cl();
    println!("원본 x={}, flag={}", x, flag); // ✅ Copy이므로 원본 유효

    // ━━ non-Copy 타입 ━━
    let name = String::from("Rust");
    let cl2 = move || println!("name={}", name); // name 소유권 이전
    cl2();
    // println!("{}", name); // ❌ 소유권 이전됨

    // ━━ 스레드: 항상 move 필요 ━━
    // Copy 타입도 스레드에서는 move로 복사본 전달
    let threshold: i32 = 60;
    let h = thread::spawn(move || {
        // threshold는 Copy이므로 복사본이 스레드로 전달
        println!("스레드에서 threshold={}", threshold);
    });
    h.join().unwrap();
    println!("메인에서도 threshold={}", threshold); // ✅

    // ━━ move와 트레이트 결정의 관계 ━━
    // move는 "캡처 방식"을 강제하지만, 트레이트(Fn/FnMut/FnOnce)는
    // 클로져 본문에서 캡처한 값을 '어떻게 사용하는가'로 결정됨

    let s = String::from("hello");
    // move로 소유권 이전했지만 내부에서 읽기만 → Fn
    let fn_with_move = move || println!("{}", s); // Fn (읽기만)
    fn_with_move();
    fn_with_move(); // ✅ Fn이므로 재호출 가능

    let v = vec![1, 2, 3];
    // move로 소유권 이전했고 내부에서 drop 사용 → FnOnce
    let fnonce_with_move = move || drop(v); // FnOnce (소비)
    fnonce_with_move();
    // fnonce_with_move(); // ❌ FnOnce
}
상황캡처 타입move 후 원본트레이트 결정
Copy 타입, 읽기만i32, bool 등✅ 유효Fn
Copy 타입, 수정i32, bool 등✅ 유효 (but 빌림)FnMut
non-Copy, move+읽기String, Vec 등❌ 이동됨Fn
non-Copy, move+수정String, Vec 등❌ 이동됨FnMut
non-Copy, move+소비String, Vec 등❌ 이동됨FnOnce

Box<dyn Fn> vs impl Fn — 동적 vs 정적 디스패치

impl Fn은 컴파일 시점에 구체적인 타입이 결정되는 정적 디스패치(static dispatch)입니다. Box<dyn Fn>은 런타임에 vtable을 통해 호출하는 동적 디스패치(dynamic dispatch)입니다. 서로 다른 타입의 클로져를 같은 컬렉션에 저장하거나 반환 타입이 런타임에 결정될 때는 Box<dyn Fn>을 사용합니다.

✅ impl Fn 사용 시기
함수의 반환 타입이 컴파일 시점에 확정될 때. 성능이 중요할 때. 단일 클로져 반환 시. 제네릭 함수 바운드.
📌 Box<dyn Fn> 사용 시기
서로 다른 클로져를 Vec 등에 저장할 때. 런타임에 클로져 결정 시. 재귀 구조. 트레이트 객체가 필요할 때.
dyn_fn_vs_impl_fn.rsBox<dyn Fn>
// ─────────────────────────────────────────────────────────
// impl Fn: 정적 디스패치 — 컴파일 시점에 타입 확정
//   장점: 인라이닝 가능, 런타임 오버헤드 없음
//   단점: 다른 타입 클로져를 같은 변수에 저장 불가
//
// Box<dyn Fn>: 동적 디스패치 — vtable을 통한 런타임 호출
//   장점: 다양한 타입의 클로져를 균일하게 처리
//   단점: 힙 할당 + 간접 호출 비용
// ─────────────────────────────────────────────────────────

// ━━ impl Fn: 정적 디스패치 ━━
fn make_adder_impl(n: i32) -> impl Fn(i32) -> i32 {
    move |x| x + n // 컴파일 시 구체적 타입 결정 → 인라이닝 가능
}

// ━━ Box<dyn Fn>: 동적 디스패치 ━━
fn make_adder_dyn(n: i32) -> Box<dyn Fn(i32) -> i32> {
    Box::new(move |x| x + n) // 힙 할당 + vtable
}

// Box<dyn Fn>의 핵심 장점: Vec에 다양한 클로져 저장
fn make_validator(flag: bool) -> Box<dyn Fn(i32) -> bool> {
    // 런타임 조건에 따라 다른 클로져 반환 → impl Fn 불가, Box 필수
    if flag {
        Box::new(|x| x > 0)  // 양수 검사
    } else {
        Box::new(|x| x < 0)  // 음수 검사
    }
}

fn main() {
    let add5_impl = make_adder_impl(5);
    let add5_dyn  = make_adder_dyn(5);
    println!("impl: {}, dyn: {}", add5_impl(10), add5_dyn(10)); // 15, 15

    // 런타임 선택
    let check_pos = make_validator(true);
    let check_neg = make_validator(false);
    println!("5 양수: {}, -3 음수: {}", check_pos(5), check_neg(-3));

    // Vec에 서로 다른 클로져 저장 (Box<dyn Fn> 필수)
    let rules: Vec<Box<dyn Fn(i32) -> bool>> = vec![
        Box::new(|x| x > 0),
        Box::new(|x| x % 2 == 0),
        Box::new(|x| x < 100),
    ];
    let n = 42;
    let all_pass = rules.iter().all(|rule| rule(n));
    println!("{}: 모든 규칙 통과={}", n, all_pass); // 42: true

    // FnMut, FnOnce도 동일 패턴
    let mut actions: Vec<Box<dyn FnMut()>> = vec![
        Box::new(|| println!("Action A")),
        Box::new(|| println!("Action B")),
    ];
    for action in &mut actions { action(); }
}
